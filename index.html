Supabase database

Setup: Create a free account at supabase.com. Create a new project, then run the SQL below in the Supabase SQL editor to create the reports table. Then paste your Project URL and anon key here.

Project URLFrom Supabase → Settings → API
Anon (public) keyFrom Supabase → Settings → API
SQL to create table ▾ 
CREATE TABLE hazard_reports (
  id          uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  ref         text NOT NULL,
  timestamp   timestamptz DEFAULT now(),
  type        text,
  location    text,
  severity    text,
  description text,
  action      text,
  photo_desc  text,
  status      text DEFAULT 'Open'
);

-- Allow anonymous inserts (public submissions)
ALTER TABLE hazard_reports ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Anyone can insert"
  ON hazard_reports FOR INSERT
  WITH CHECK (true);

CREATE POLICY "Anon key can read"
  ON hazard_reports FOR SELECT
  USING (true);

CREATE POLICY "Anon key can update status"
  ON hazard_reports FOR UPDATE
  USING (true);
